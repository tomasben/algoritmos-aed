## 4.01
Diseñar un algoritmo para acceder, eliminar o insertar el k-ésimo elemento de
una lista (siendo k una posición dada). Si la lista está vacía o si el valor
de k esta fuera del rango del índice de la lista, invocar al procedimiento de
ERROR. En cualquier otro caso, efectuar el procedimiento solicitado.

```
ACCION ejercicio (prim: puntero a Nodo) ES
  AMBIENTE
    Nodo = REGISTRO
      dato: entero
      prox: puntero a Nodo
    FIN_REGISTRO

    prim, p, a, q: puntero a nodo
    k, opcion, iter: entero

    PROCEDIMIENTO acceder(k: entero) ES
      MIENTRAS (p <> nil) Y (iter <> k) HACER
        p := *p.prox
        iter := iter + 1
      FIN_MIENTRAS

      SI (*p.dato <> nil) ENTONCES
        ESCRIBIR("El valor en la posición N° ", k, " es: ", *p.dato)
      CONTRARIO
        ESCRIBIR("ERROR: la lista tiene ", iter, " elementos.")
      FIN_SI
    FIN_PROCEDIMIENTO

    PROCEDIMIENTO eliminar(k: entero) ES
      MIENTRAS (p <> nil) Y (iter <> k) HACER
        a := p
        p := *p.prox
      FIN_MIENTRAS

      SI (q = NIL) ENTONCES
        ESCRIBIR("ERROR: índice fuera de rango de lista: ", k)
      CONTRARIO
        *a.prox = *q.prox
        DISPONER(q)
      FIN_SI
    FIN_PROCEDIMIENTO

    PROCEDIMIENTO insertar(k: entero) ES
      MIENTRAS (p <> nil) Y (iter <> k) HACER
        a := p
        p := *p.prox
      FIN_MIENTRAS

      SI (q = NIL) ENTONCES
        ESCRIBIR("ERROR: índice fuera de rango de lista: ", k)
      CONTRARIO
        NUEVO(q)
        ESCRIBIR("Ingrese el nuevo dato a insertar: ")
        LEER(*q.dato)

        *q.prox := p
        *a.prox := q
      FIN_SI
    FIN_PROCEDIMIENTO
  PROCESO
    p := prim
    iter := 1

    ESCRIBIR("Ingrese la posición de un elemento a acceder, eliminar o insertar: ")
    LEER(k)

    ESCRIBIR("OPCIONES: ")
    ESCRIBIR("  [1] acceder elemento")
    ESCRIBIR("  [2] eliminar elemento")
    ESCRIBIR("  [3] insertar elemento")
    LEER(opcion)

    SI (q = nil) ENTONCES
      ESCRIBIR("ERROR: lista vacía")
    CONTRARIO
      SEGUN opcion HACER
        1: acceder(k)
        2: eliminar(k)
        3: insertar(k)
      FIN_SEGUN
    FIN_SI
FIN_ACCION
```

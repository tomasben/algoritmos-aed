## 4.08
Escribir un algoritmo que permita buscar, insertar o borrar un elemento
identificado con una clave determinada en una lista circular simplemente
encadenada.

```
ACCION ejercicio ES
  AMBIENTE
    Objeto = REGISTRO
      clave: entero
      ..otros campos
    FIN_REGISTRO

    Nodo = REGISTRO
      obj: Objeto
      prox: puntero a Nodo
    FIN_REGISTRO

    reg: Objeto
    prim, p, a, q: puntero a Nodo
    buscado, eleccion: entero

    PROCEDIMIENTO mostrar_menu() ES
      ESCRIBIR("Opciones:")
      ESCRIBIR("  [1] Buscar un elemento")
      ESCRIBIR("  [2] Insertar un elemento")
      ESCRIBIR("  [3] Eliminar un elemento")
    FIN_PROCEDIMIENTO

    PROCEDIMIENTO buscar() ES
      MIENTRAS (*p.prox <> prim) Y (*p.obj.clave <> buscado) HACER
        a := p
        p := *p.prox
      FIN_MIENTRAS

      SI (*p.prox = prim) Y (*p.obj.clave <> buscado) ENTONCES
        ESCRIBIR("ERROR: el objeto no se encuentra en la lista.")
      CONTRARIO
        ESCRIBIR("El objeto solicitado es: ")
        ..mostrar registro campo por campo
      FIN_SI
    FIN_PROCEDIMIENTO

    PROCEDIMIENTO insertar() ES
      ESCRIBIR("Ingrese el objeto a insertar: ")
      reg.clave := buscado
      ..leer campo por campo

      NUEVO(q)
      *q.obj := reg

      MIENTRAS (*p.prox <> prim) Y (*p.obj.clave < buscado) HACER
        a := p
        p := *p.prox
      FIN_MIENTRAS

      SI (*p.prox = prim) Y (*p.obj.clave > buscado) ENTONCES
        *p.prox := q
        *q.prox := prim
      CONTRARIO
        *a.prox := q
        *q.prox := p
      FIN_SI
    FIN_PROCEDIMIENTO

    PROCEDIMIENTO eliminar() ES
      MIENTRAS (*p.prox <> prim) Y (*p.obj.clave <> buscado) HACER
        a := p
        p := *p.prox
      FIN_MIENTRAS

      SI (*p.prox = prim) Y (*p.obj.clave <> buscado) ENTONCES
        ESCRIBIR("No se encontró el objeto a eliminar.")
      CONTRARIO
        SI (*p.prox = prim) ENTONCES
          *a.prox := prim
        CONTRARIO
          *a.prox := *p.prox
        FIN_SI
        DISPONER(p)
      FIN_SI
    FIN_PROCEDIMIENTO
  PROCESO
    p := prim

    mostrar_menu()
    ESCRIBIR("Ingrese la operación a realizar:")
    LEER(eleccion)
    ESCRIBIR("Ingrese la clave del objeto: ")
    LEER(buscado)

    SEGUN eleccion HACER
      1: buscar()
      2: insertar()
      3: eliminar()
    FIN_SEGUN
FIN_ACCION
```

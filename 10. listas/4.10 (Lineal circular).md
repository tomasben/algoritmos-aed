## 4.10
Se dispone de una lista circular con un conjunto de números naturales. Este
conjunto está dividido en 6 grupos. En cada grupo el primer elemento indica la
cantidad de números subsiguientes a él que integran dicho grupo. Se desea
generar una nueva lista doblemente encadenada en la cual cada nodo contendrá el
promedio de cada grupo. Dicha lista debe quedar ordenada en forma ascendente y,
al final se debe informar cuál fue el máximo y el mínimo promedio de todo el
conjunto de números.

```
ACCION ejercicio(prim_conj: puntero a Conjunto) ES
  AMBIENTE
    Conjunto = REGISTRO
      num: entero
      prox: puntero a Conjunto
    FIN_REGISTRO

    Promedio = REGISTRO
      prom: real
      ant, prox: puntero a Promedio
    FIN_REGISTRO

    t: puntero a Conjunto
    prim, ult, p, a, q: puntero a Promedio

    i, cant: entero
    prom, max_prom, min_prom: real
  PROCESO
    ult := prim
    max_prom := LV
    min_prom := HV
    t := prim_conj

    PARA i := 1 HASTA 6 HACER
      prom := 0
      cant := *t.num

      PARA j := 1 HASTA cant HACER
        t := *t.prox
        prom := prom + *t.num
      FIN_PARA

      prom := prom / cant
      SI (prom > max_prom) ENTONCES
        max_prom := prom
      CONTRARIO
        SI (prom < min_prom) ENTONCES
          min_prom := prom
        FIN_SI
      FIN_SI

      NUEVO(q)
      *q.prom := prom

      p := prim
      a := nil
      MIENTRAS (p <> nil) Y (*p.prom < prom) HACER
        a := p
        p := *p.prox
      FIN_MIENTRAS

      SI (prim = nil) ENTONCES
        *q.ant := nil
        *q.prox := nil
        prim := q
        ult := q
      CONTRARIO
        SI (*q.prom < *prim.prom) ENTONCES
          *prim.ant := q
          *q.ant := nil
          *q.prox := prim
          prim := q
        CONTRARIO
          SI (p = nil) ENTONCES
            *q.prox := nil
            *q.ant := a
            *a.prox := q
            ult := q
          CONTRARIO
            *a.prox := q
            *q.ant := a
            *q.prox := p
            *p.ant := q
          FIN_SI
        FIN_SI
      FIN_SI

      t := *t.prox
    FIN_PARA

    ESCRIBIR("El mayor promedio de los 6 grupos fue de: ", max_prom)
    ESCRIBIR("Y el menor promedio de los 6 grupos fue de: ", min_prom)
FIN_ACCION
```
